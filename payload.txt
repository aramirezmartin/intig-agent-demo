function F {
    param (
        [byte[]]$x,
        [string]$y
    )

    $s = 0..255
    $j = 0
    $k = [byte[]][System.Text.Encoding]::ASCII.GetBytes($y)

    for ($i = 0; $i -lt 256; $i++) {
        $j = ($j + $s[$i] + $k[$i % $k.Length]) % 256
        $temp = $s[$i]
        $s[$i] = $s[$j]
        $s[$j] = $temp
    }

    $i = 0
    $j = 0
    $output = New-Object byte[] ($x.Length)
    for ($m = 0; $m -lt $x.Length; $m++) {
        $i = ($i + 1) % 256
        $j = ($j + $s[$i]) % 256
        $temp = $s[$i]
        $s[$i] = $s[$j]
        $s[$j] = $temp
        $output[$m] = $x[$m] -bxor $s[($s[$i] + $s[$j]) % 256]
    }

    return $output
}
